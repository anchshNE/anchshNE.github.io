第4关 练习任务03-05-04 四个数中的最大值

 
#include


int main()
{
    int a, b, c, d;
    scanf("%d %d %d %d", &a, &b, &c, &d);
    printf("%d", (a < b ? b : a) < (c < d ? d : c) ? (c < d ? d : c) : (a < b ? b : a));

    return 0;
}

第6关 练习任务03-05-06 苹果和虫子

 
#include

int main()
{
    int n, x, y;
    scanf("%d %d %d", &n, &x, &y);

    printf("%d", n - ((y%x) == 0 ? y/x : (y/x)+1));
    return 0;
}
