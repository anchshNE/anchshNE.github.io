第1关 引导任务03-07-01 数学函数

 
#include
#include
#define PI 3.1415926
int main(){  
     double a,b;
     scanf("%lf%lf",&a,&b); //输入两个实数
     printf("sin(%lf)=%lf\n",a,sin(a*PI/180));//输出a的正弦,a*PI/180为弧度
     printf("cos(%lf)=%lf\n",a,cos(a*PI/180));//输出a的余弦,a*PI/180为弧度
     printf("exp(%lf)=%lf\n",a,exp(a));//输出e的a次方 
     printf("log(%lf)=%lf\n",a,log(a));//输出以2为底a的对数 
     printf("log10(%lf)=%lf\n",a,log10(a));    //输出以10为底a的对数
     printf("pow(%lf,%lf)=%lf\n",a,b,pow(a,b));//输出a的b次方 
     return 0;
}
