第3关 练习任务04-05-03 辛巳蛇宝男

 
#include

int main()
{
    int year, month, day, gender;
    scanf("%*6d%4d%2d%2d%*2d%1d", &year, &month, &day, &gender);
    year == 2001 ? gender % 2 != 0 ? month == 1 ? day >= 24 ? printf("YES") : printf("NO") : printf("YES") : printf("NO") : 0;
    year == 2002 ? gender % 2 != 0 ? month <= 2 ? day <= 11 ? printf("YES") : printf("NO") : printf("NO")  : printf("NO") : 0;
    year ^ 2001 && year ^ 2002 ? printf("NO") : 0;
    return 0;
}

{
    int a, b;
    scanf("%d*%d", &a, &b);
    a = a % 10 + (a % 1000 - a % 100) / 10;
    b = (b % 100 - b % 10) / 10 + (b % 10000 - b % 1000) / 100;
    printf("%d*%d=%d", a, b, a * b);
